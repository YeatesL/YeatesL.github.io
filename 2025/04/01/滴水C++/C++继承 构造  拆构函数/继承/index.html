继承

//———继承 本质：—–数据的复制—–
//怎么用其他函数的成员
//无变化 能用的原因：同一个指针

struct person
{
int age;
int sex；
};
//——-当需要应用其他函数的成员时，例：（struct teather:person） 表示需要person函数里的成员
struct teather:person
{
int level;
int classId;
};
//——-例：（struct student:person) 表示需要person函数里的成员,,,即为–继承–
struct student:person
{
int code;
int score;
};
void Test()
{
student t;

t.age = 1;
t.sex = 2;
t.code = 3;
t.score = 4;

}

//—–通过指针来访问里面的成员
void Test()
{
student t;

t.age = 1;
t.sex = 2;
t.code = 3;
t.score = 4;

student* pt = &t;		//用指针来访问	
printf("%d",pt->code);

person* pt = &t;		//能这样是因为它（子类对象）包含父类成员----(struct student:person)----
printf("%d",pt->age);		//父类的指针也可以访问
printf("%d",pt->code);		//这个是不行的   因为person内没有包含 student函数的东西
    //但  +1  就能了		因为继承本质就是复制

}

struct person
{
int age;
int sex；
};
struct student:person
{
