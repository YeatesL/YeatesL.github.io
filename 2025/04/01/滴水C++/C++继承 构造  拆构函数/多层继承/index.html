多层继承

//多层继承，即 继承后，再继承
struct X
{
int a；
int b；
}；
struct Y:X
{
int a；
int d；
}；
struct Z:Y struct Z:Y:X //要使用 a 的话，也与下边的一样，要表明是X的a，还是Y的a
{
int a；
int f；
}；
——–//即使它们中间的名字有重复 分配空间的时候依然是 24字节
void Test()
{
Z z;

z.X::a = 1;		//名字相同时要准确的标出谁的a，是X的还是Y的
z.Y::a = 3;
z.Z::a = 6;

}
//它们的反汇编代码一样， 原因就是 继承—–复制

//微软（不建议多重继承）；继承 一个直接父类只有一个
