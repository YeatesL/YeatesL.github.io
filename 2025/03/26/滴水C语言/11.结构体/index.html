11.结构体

struct xxx //定义一个结构体
{
int x;
int y;
};
}S1,S2; //用结构体定义两个结构体变量S1,S2\

struct Stu
{
int num;
cgar name[20];
char sex;
int age;
float score;
}s[30]; //定义了有30个变量的结构体数组

Stu* p; //指针变量*p /指向结构体数组的指针
p = s; //指针变量p指向结构体数组的首地址

p->num; //使用p指向结构体数组的元素
