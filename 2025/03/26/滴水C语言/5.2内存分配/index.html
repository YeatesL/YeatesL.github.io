int* ptr=(int*)malloc(sizeof(int)*128);

if(ptr=NULL)
{
return 0;
}
//初始化分配到内存空间
memset(ptr,0,sizeof(int)*128)
*ptr=1; //使用指针
free(ptr); //使用完毕，释放申请的堆空间
ptr = NULL; //将指针设置为NULL,方便下次使用
