C++构造 --重载

/*构造函数
1.没有返回值，即（void）
2与结构的名字要一样（与类同名）
3.是在创造对象的时候执行的。
4.主要用于初始化数据
5.多个，成为重载
6.编译器不要求必须提供
*/
在开始的时候，对对象进行赋值
struct person
{
int age;
int level;
//———定义了构造函数就要用：//person p(1,2)；//
//构造函数： 3.是在创造对象的时候执行的。
person(int age,int level) //这个是函数，但又有区别 1.没有返回值，即（void）2.与结构的名字要一样（与类同名）
{
this->age = age;
this->level = level;
}
//——-多个构造函数 叫重载
person()
{
//不想赋值了，可以在写上这个（前提是不能和前面函数的一样），保证了person ()不会报错
}

//——普通成员的—重载
//——-通过对象的方式，调用方法
viod print()
{
printf(“%d %d”,age,level);
}
viod print(int x) //viod print(char x)
{
printf(“%d %d”,age,level);
}
//——-C++里允许函数名一样，但要保证参数，类型，个数，不一样就行 —-但返回类型要一样（void，int）

};
int main( )
{
person p(1,2)； //在栈里边分配空间，用完后就是垃圾

p.print();

}

析构函数

/*析构函数：
1. 做收尾的，主要用于清理工作
2.析构函数不允许重载，它只有一个
3.不能带参数
4.不能带返回值
5.编译器不要求必须提供

/
//—-栈 全局变量 堆
struct person
{
int age;
int level;
char arr;

person()
{
    this->age = age;
    this->level = level;
    arr= (char*)malloc(1024);		//在堆中申请空间  要自己对堆里的东西释放
}

//—–对象不用了，要释放
~person()
{
free(arr);
}
};

3.1指针

（加减的时候要砍）
例
char 类型 是把值取出来，然后把取出来的值当成地址，再次取值
若是char** 类型 （在上面的基础上）再次把取出来的值当成地址，再次取值
（以此类推）
char* p
p==（p+0）==p[0]
char p
（（p+2）+1）==p[2][1]
char*** p
（（*（p+1）+2）+3）==p[1][2][3]

char** p		p为char(*)类型	(*)p为char*类型	(**)p为char类型
*(*(p+1)+1)   p为char**  p+1为char**  四字节
（p+1）p为char**，+1时要砍*，后为char* 为四字节 //所以（p+1）中的+1实际上是加上四字节.
  (*(p+1）+1)   *(p+1)为char* 类型，+1时要砍*，后为char类型，为一字节 //所以(*(p+1）+1) 中的后一个+1实际上为加上一字节。

int i=100; //把值存在i里
int* p=&i; //lea 取i的地址存在p里
printf(“%d\n”, p); //实际上是i的地址
printf(“%d\n”, *p); //取的100（p里存着的i地址里面的值）

//指针 *
{
结构体指针 x—>结构体（x里存得地址，地址指向的结构体）
指针的指针 x—>指针 （x里存得地址，地址指向的指针）
指针数组 x—>数组 （x里存得地址，地址指向的数组）
}
//指针函数 是函数 本质是返回值是 指针。

int（px）[5] //px int[5]
++ 相当于px砍掉，相当于int[5] (五个int）也就是长度是五个int 也就是 4*5=20

int arr[20]={1,2,3,4,5,……….};
int (px)[2];
px=(int ()[2])arr; //取得是arr[0]的地址
printf(“%d\n”,((px));
printf(“%d\n”,((px+0)+0);
printf(“%d\n”,((px+1)+1); –px[1][1] //px是int()[2]类型，把砍掉成 int[2] 相当于加了八个字节。4 2 = 8 也就是从arr[0]的地址转移到了arr[2]
//把(px+1) 弄完再+1成了数组—>成了int类型，+1 也就是 加上四个字节
printf(“%d\n”,((px+3)+3); —px[3][3] —-> printf出的值 –10
//貌似好像会报错…

px=(int ()[2])arr;
printf(“%d %d\n”,px,px); //这里px与px值一样，但它俩的类型完全不同
//理解:px指向了一个储存了数组首地址的地址，而px就是获取这个地址的值，这个地址的值就是数组首地址的值

（要带本身字节宽度）
//二维指针
char (p)[2][3] // 正常p的宽度：123 （1是因为是char–一字节）
printf(“%d\n”,((((p+2)+3)+4)); // 26 133 +41
//2(23) (13*3) [char(1) 乘4]
